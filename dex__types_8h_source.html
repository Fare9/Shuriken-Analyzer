84 /// @brief Operator == for comparison between two DVMTypes
85 inline bool operator==(const DVMType &lhs, const DVMType &rhs) {
86 if ((lhs.get_type() == rhs.get_type()) && (lhs.get_raw_type() == rhs.get_raw_type()))
87 return true;
88 return false;
89 }
