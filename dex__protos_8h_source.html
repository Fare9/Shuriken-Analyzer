9/*
10 * Each proto structure is:
11 * ProtoID {
12 * uint shorty_idx, # OFFSET to a string with return type

14 * uint parameters_off # ---------| Offset to parameters in type list type
15 * } |
16 * |
17 * ------------------------------------
18 * |
19 * v
20 * type_list {
21 * uint size, # Size of the list in entries
22 * list type_item[size] # ---------|
23 * } |
24 * |
25 * -------------------------------------
26 * |
27 * v
28 * type_item {
29 * ushort type_idx # type_idx of each member
30 * }
31 *
32 * ProtoID[] protos
33 */
